let a=10;
let b=10;

let c=10;

var e=15;
var f="15"
console.log(e==f); // "The doubleEqualTo(==) is not check the datatype[i.e, string or number] It only check the value(here number)" OUTPUT:true

var e=15;
var f="15";
console.log(e===f); // "The tripleEqualTo also checks the DataType and value ,suppose the data-type is vary it give the output false" OUTPUT:false

var g=10;
var h=10;
console.log(g!=h); // "The variable g & h are equal" OUTPUT:false

var g=10;
var h=15;
console.log(g!=h); // "The variable g & h are not equal" OUTPUT:true
