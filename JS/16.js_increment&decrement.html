Increment & Decrement

Postfix Increment


var a=5;
b=a++;
console.log("value of B:",b);
console.log("value of a:",a);

OUTPUT:value of B: 5
value of a: 6

Prefix Increment


var a=5;
b=++a;
console.log("value of b:",b);

OUTPUT: value of b: 6

postfix Decrement


var a=5;
b=a--;
console.log("value of B:",b);
console.log("value of a:",a);

OUTPUT: value of B: 5
value of a: 4

Prefix Decrement


var a=5;
b=--a;
console.log("value of b:",b);

OUTPUT: value of b: 4
