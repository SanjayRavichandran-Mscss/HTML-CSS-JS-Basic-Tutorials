LOGICAL OPERATOR

  • Logical and(&&)
  • Logical or(||)
  • Logical not(!)

AND operator

var mark=40;
console.log("mark=40");

console.log(mark>35 && mark<100) //OUTPUT :true

var mark=30;
console.log("mark=30");

console.log(mark>35 && mark>100) //OUTPUT false

var mark=300;
console.log(mark>35 && mark<100) //OUTPUT false

OR operator

let a=10;
console.log(a==20 || a==40 || a==50 || a==10 || a==30) // OUTPUT:true

NOT operator

var b=20;
console.log(b!==20); //OUTPUT:false

var b=30;
console.log(b!==20); //OUTPUT:true
