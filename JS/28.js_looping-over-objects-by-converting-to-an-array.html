Looping over objects by converting to an array

let user = {
name: "sanjay",
Email: "sanjay@gmail.com",
password : 123456,
phone_number: 987654321,
};

let array_keys = Object.keys(user); Here , Object.keys(user) is convert the object form into array form, and stored into the corresponding variablearray_keys.
console.log(array_keys);

let array_values = Object.values(user);
console.log(array_values); Here , Object.values(user) is convert the object form into array form, and stored into the corresponding variable array_values.

for(i=0;i<=array_keys.length;i++)
{
console.table(array_keys[i] + " : " +array_values[i]);
}
