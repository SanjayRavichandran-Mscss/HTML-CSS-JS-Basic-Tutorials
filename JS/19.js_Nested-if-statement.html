mat=99,phy=98,che=99;
total=mat+phy+che;
avg=total/3;

if(mat>=35 && phy>=35 && che>=35 )
{
console.log("RESULT : PASS");

if(avg>90 && avg<=100)
{
console.log("GRADE :A");
}

else if(avg>80 && avg<=90)
{
console.log("GRADE :B");
}

else if(avg>70 && avg<=80)

log("GRADE :C");
}
else if(avg>60 && avg<=70)
{
console.log("GRADE :D");
}

else if(avg>50 && avg<=60)
{

}

else
{

}

}

else
