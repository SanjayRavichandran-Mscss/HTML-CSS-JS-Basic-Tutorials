RELATIONAL OPERATOR

  • greater than (>)
  • less than (<)
  • greater than or equal to(>=)
  • less than or equal to(<=)
var a=50,b=30;
console.log("a=50,b=30");

console.log("a greater than b: ",a>b); //OUTPUT:true
console.log("a lesser than b: ",a<b); //OUTPUT:false

var a=20,b=20;
console.log("a=20,b=20");

console.log("a greater than or equalto b",a>=b); //OUTPUT:true
console.log("a lesser than or equalto b",a<=b); //OUTPUT:true
