ASSIGNMENT OPERATOR

var a=10;
console.log(a); // a=10

OUTPUT:10


a=a+5; // 10+5=15
console.log(a); //a=15

OUTPUT:15


a+=5; // a=a+5; , a=15+5
console.log(a); // a=20

OUTPUT:20


a-=5; //a=a-5; , 20-5=15
console.log(a); // a=15

OUTPUT:15


a*=5; //a=a*5; a=15*5;
console.log(a); // a=75;

OUTPUT:75


a/=15; //a=a/5; a=75/5;
console.log(a); // a=15;

OUTPUT:15


a**=5; //a=a**5; a=15**5;
console.log(a); //a=759375;

OUTPUT:759375


a%=5; //a=a%5; a=759375%5;
console.log(a); // a=0;

OUTPUT:0
